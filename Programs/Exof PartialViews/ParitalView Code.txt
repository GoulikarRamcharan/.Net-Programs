Step 2 
Create Empty Controller Called Product Controller
And Write Following Code 

using Microsoft.AspNetCore.Mvc;
using PartialViewInMVC.Models;

namespace ExofPartialView.Controllers
{
    public class ProductController : Controller
    {
        private List<Product> products = new List<Product>();
        public ProductController()
        {
            products = new List<Product>()
            {
                new Product { ProductID =1, Name ="Colgate", Category = "Tooth Paste", Price = 10},
                new Product { ProductID =2, Name ="Close Up", Category = "Tooth Paste", Price = 20},
                new Product { ProductID =3, Name ="Rin", Category = "Detergent", Price = 30},
                new Product { ProductID =4, Name ="Surf", Category = "Detergent", Price = 40},
                new Product { ProductID =5, Name ="Cadbury", Category = "Choclate", Price = 50},
                new Product { ProductID =6, Name ="Kitkat", Category = "Choclate", Price = 50}
            };
        }

        //Product/Details/1
        public ActionResult Index()
        {
            var ProductDetails = products.ToList();
            return View(ProductDetails);
        }
    }
}
